data type conversion

reference: https://www.geeksforgeeks.org/data-type-conversion-in-r/

# value assignment 
name <- "GeeksforGeeks"
age <- 20
pwd <- FALSE
  
# type checking 
typeof(name)
[1] "character"
typeof(age)
[1] "double"
typeof(pwd)
[1] "logical"
# value assignment 
age <- 20 
pwd <- "FALSE"
  
# Converting type 
as.logical(age)
[1] TRUE
as.logical(pwd)
[1] FALSE
# value assignment 
age <- 20 
pwd <- FALSE

# Converting type 
as.character(age)
[1] "20"
as.character(pwd)
[1] "FALSE"
# value assignment 
age <- "20"
pwd <- FALSE

# Converting type 
as.numeric(age)
[1] 20
as.numeric(pwd)
[1] 0
